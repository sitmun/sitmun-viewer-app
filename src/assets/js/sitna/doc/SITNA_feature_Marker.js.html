/**
 * Asigna estilos al marcador.
 * @method setStyle
 * @memberof SITNA.feature.Marker
 * @instance
 * @param {SITNA.feature.MarkerStyleOptions} style - Objeto de opciones de estilo de marcador.
 * @returns {SITNA.feature.Marker} La propia entidad geográfica.
 */

class Marker extends Point {
    STYLETYPE = 'marker';

    constructor(coords, options) {
        super(coords, options);
        const self = this;

        if (!self.wrap.isNative(coords)) {
            const locale = self.layer && self.layer.map ? self.layer.map.options.locale : Cfg.locale;
            self.title = self.options.title || TC.i18n[locale][Consts.MARKER];
            self.wrap.createMarker(coords, options);
        }
    }
}

export default Marker;
